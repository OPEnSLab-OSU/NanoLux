120inline void handle_pattern_update_put_request(AsyncWebServerRequest* request, JsonVariant& json) {
121 if (request->method() == HTTP_PUT) {
122 const JsonObject& payload = json.as<JsonObject>();
123
124 int status = HTTP_OK;
125
126 uint8_t count = payload["subpattern_count"];
127 uint8_t alpha = payload["alpha"];
128 uint8_t mode = payload["mode"];
129 uint8_t noise = payload["noise"];
130
131 bound_byte(&count, 1, 4);
132 bound_byte(&alpha, 0, 255);
133 bound_byte(&mode, 0, 1);
134 bound_byte(&noise, 0, 100);
135
136 if(count != loaded_pattern.subpattern_count)
137 pattern_changed = true;
138
139 if(mode != loaded_pattern.mode)
140 pattern_changed = true;
141
142 loaded_pattern.subpattern_count = count;
143 loaded_pattern.alpha = alpha;
144 loaded_pattern.noise_thresh = noise;
145 loaded_pattern.mode = mode;
146
147 manual_control_enabled = false;
148
149 request->send(
150 HTTP_OK,
151 CONTENT_TEXT,
152 build_response(
153 true,
154 "success",
155 nullptr));
156 } else {
157 request->send(HTTP_METHOD_NOT_ALLOWED);
158 }
159}
